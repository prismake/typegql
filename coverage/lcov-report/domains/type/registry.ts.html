import { ObjectTypeError } from './error';
 
export const objectTypeSymbol = Symbol('objectType');
 
declare global {
  interface Function {
    [objectTypeSymbol]?: TypeConfig;
  }
}
 
export interface TypeConfig {
  name: string;
  description: string;
}
 
export function registerObjectType(targetClass: Function, options: TypeConfig) {
  targetClass[objectTypeSymbol] = options;
}
 
export function isTypeRegistered(targetClass: any): targetClass is { new (): any } {
  if (typeof targetClass !== 'function') {
    return false;
  }
  return !!targetClass[objectTypeSymbol];
}
 
export function getObjectTypeConfig(targetClass: Function) {
  const config = targetClass[objectTypeSymbol];
 
  if (!config) {
    throw new ObjectTypeError(
      targetClass,
      `Class is not registered. Make sure it's decorated with @Type decorator`,
    );
  }
  return config;
}
